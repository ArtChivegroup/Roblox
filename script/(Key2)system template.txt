local keyFile = "bloxhub.key"

local hour = 60 * 60

-- FUNCTIONS
local function fetchKey()
    local success, response = pcall(function()
        return game:HttpGet(keyURL)
    end)
    if success then
        return string.match(response, "%S+")
    else
        warn("Failed to fetch key!")
        return nil
    end
end

local function getSavedKey()
    if isfile and isfile(keyFile) and isfile(timeFile) then
        local savedTime = tonumber(readfile(timeFile))
        if savedTime and (os.time() - savedTime) < hour then
            return readfile(keyFile)
        end
    end
    return nil
end

local function saveKey(key)
    if writefile then
        writefile(keyFile, key)
        writefile(timeFile, tostring(os.time()))
    end
end

-- MAIN EXECUTE (GUI ONLY IF NEEDED, AUTO HWID CHECK)
local function mainScript()
    -- Ganti dengan script utama kamu
    print("Key system passed. Main script would load here.")
    loadstring(game:HttpGet("url"))()

end

local savedKey = getSavedKey()
if savedKey then
    -- Tampilkan GUI sebentar untuk checking HWID, lalu langsung hilang jika key valid
    local gui, checkingLabel, keyInput = createUI(mainScript)
    checkingLabel.Text = "Checking your key HWID..."
    keyInput.Text = savedKey
    task.spawn(function()
        wait(1)
        local currentKey = fetchKey()
        if savedKey == currentKey then
            StarterGui:SetCore("SendNotification", { Title = "BloxHub", Text = "Auto-Login Success! Loading...", Duration = 1 })
            saveKey(savedKey)
            wait(1)
            gui:Destroy()
            mainScript()
        else
            checkingLabel.Text = ""
            keyInput.Text = ""
        end
    end)
else
    -- Tidak ada key valid, tampilkan UI normal
    createUI(mainScript)
end
