-- LOGIC FUNCTIONS
local function fetchKey(setProgress)
    if setProgress then setProgress(0.10, "Fetching key from server...") end
    local success, response = pcall(function()
        return game:HttpGet(keyURL)
    end)
    if success then
        if setProgress then setProgress(0.35, "Key fetched successfully.") end
        return string.match(response, "%S+")
    else
        warn("Failed to fetch key!")
        if setProgress then setProgress(0, "Failed to fetch key!") end
        return nil
    end
end

local function getSavedKey(setProgress)
    if setProgress then setProgress(0.08, "Checking local key...") end
    if isfile and isfile(keyFile) and isfile(timeFile) then
        local savedTime = tonumber(readfile(timeFile))
        if savedTime and (os.time() - savedTime) < hour then
            if setProgress then setProgress(0.2, "Valid local key found.") end
            return readfile(keyFile)
        end
    end
    if setProgress then setProgress(0.08, "No valid local key found.") end
    return nil
end

local function saveKey(key, setProgress)
    if writefile then
        writefile(keyFile, key)
        writefile(timeFile, tostring(os.time()))
        if setProgress then setProgress(0.90, "Saving key...") end
    end
end

local function mainScript()
    -- LAUNCHER STYLE LOADING
    local setProgress, destroyLoading = showLauncherLoading()
    setProgress(0.15, "Fetching script...")
    local success, result = pcall(function()
        return game:HttpGet(mainScriptURL)
    end)
    if success and type(result) == "string" and #result > 0 then
        setProgress(0.8, "Executing script...")
        local ok, loadErr = pcall(function()
            loadstring(result)()
        end)
        if ok then
            setProgress(1, "Done!")
        else
            setProgress(1, "Script Error!")
            StarterGui:SetCore("SendNotification", {
                Title = "BloxHub",
                Text = "Script Error: "..tostring(loadErr),
                Duration = 5
            })
        end
    else
        setProgress(1, "Failed to fetch script!")
        StarterGui:SetCore("SendNotification", {
            Title = "BloxHub",
            Text = "Failed to fetch main script!",
            Duration = 5
        })
    end
    wait(1)
    destroyLoading()
end

local function runSystem()
    local gui, checkingLabel, keyInput, setProgress, destroyProgressBar = createUI(mainScript)
    setProgress(0.05, "Initializing key system...")
    local savedKey = getSavedKey(setProgress)
    if savedKey then
        setProgress(0.5, "Auto login, checking HWID...")
        checkingLabel.Text = "Checking your key HWID..."
        keyInput.Text = savedKey
        task.spawn(function()
            wait(1)
            setProgress(0.7, "Verifying key with server...")
            local currentKey = fetchKey(setProgress)
            if savedKey == currentKey then
                StarterGui:SetCore("SendNotification", { Title = "BloxHub", Text = "Auto-Login Success! Loading...", Duration = 1 })
                saveKey(savedKey, setProgress)
                setProgress(1, "Auto login success!")
                wait(0.5)
                gui:Destroy()
                mainScript()
            else
                checkingLabel.Text = ""
                keyInput.Text = ""
                setProgress(0, "Auto login key is not valid!")
            end
        end)
    else
        setProgress(0.2, "No valid local key, please input your key.")
    end
end

runSystem()
